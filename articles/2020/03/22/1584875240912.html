Android的各种通知Notification、Dialog、Toast、Snackbar

2020-03-22
   
Android 夜雨飘零
    0 0

有各种各样的通知，不同情况使用不同的通知方式，使用户在体验上更佳，下面就讲一下四种常用的通知方式Notification、Dialog、Toast、Snackbar

通知栏（Notification）

 1public void showNotification() {
 2        //通过Notification.Builder 创建一个notification
 3        NotificationCompat.Builder builder = new NotificationCompat.Builder(getApplicationContext());
 4
 5        //第一次收到通知时，在通知栏显示的内容
 6        builder.setTicker("中奖通知");
 7
 8        //用户点击后自动消失
 9        builder.setAutoCancel(true);
10
11        //在通知栏显示的标题
12        builder.setContentTitle("恭喜你中奖了");
13
14        //在通知栏显示的内容
15        builder.setContentText("你的号码被我公司抽中，中奖100万，请及时联络：139999");
16
17        //设置小图标，必须设置，否则通知无法显示
18        builder.setSmallIcon(R.mipmap.ic_launcher);
19
20        //设置用户无法去掉通知，只能通过代码根据id号去掉
21        //builder.setOngoing(true);
22
23        //设置静止和振动的数组，单位是毫秒，静止，振动，静止，振动···
24        builder.setVibrate(new long[]{0, 500, 500, 500});
25
26        //设置提示LED灯，颜色，亮起的时长，熄灭的市场
27        builder.setLights(Color.BLUE, 200, 200);
28
29        //设置通知的声音，要根据自己的音频位置
30        builder.setSound(Uri.fromFile(new File("/system/media/audio/ui/usb_effect.ogg")));
31
32        //设置默认的通知铃声、振动、LED灯
33        //builder.setDefaults(NotificationCompat.DEFAULT_ALL);
34
35        //设置优先级
36        builder.setPriority(NotificationCompat.PRIORITY_MAX);
37
38        Notification notification = builder.build();
39        NotificationManager manager = (NotificationManager) getSystemService(Context.NOTIFICATION_SERVICE);
40        //第一个参数是用于取消通知的id号
41        manager.notify(1, notification);
42    }

吐司（Toast）

 1private Toast toast;
 2
 3    public void showToast(String content) {
 4        //判断是否为空，避免生成多个Toast
 5        if (toast == null) {
 6            toast = Toast.makeText(this, content, Toast.LENGTH_SHORT);
 7        } else {
 8            toast.setText(content);
 9        }
10        toast.show();
11    }
