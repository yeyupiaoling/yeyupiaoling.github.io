Android网络框架OKhttp3学习笔记

OKhttp3是一个非常强大的Android网络框架，它是由Square公司开发并开源的，很大Android开发者都会使用到，所以我也要来学学。

服务器

为了方便测试，我们需要一个后台服务器的的应用，下面是一个Java Web的Servlet，它的功能是接收客户端发来的登录数据，判断密码是否正确，并返回结果（JSON格式）

 1import javax.servlet.ServletException;
 2import javax.servlet.annotation.WebServlet;
 3import javax.servlet.http.HttpServlet;
 4import javax.servlet.http.HttpServletRequest;
 5import javax.servlet.http.HttpServletResponse;
 6import java.io.IOException;
 7import java.io.OutputStreamWriter;
 8import java.io.PrintWriter;
 9import java.io.Writer;
10
11@WebServlet(name = "MyServlet",urlPatterns = {"/MyServlet"})
12public class MyServlet extends HttpServlet {
13    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
14        //获取表单数据
15        String number = request.getParameter("number");
16        String pwd = request.getParameter("pwd");
17        String body;
18        //判断密码和账号有没有正确
19        if (number.equals("12345") && pwd.equals("12345")){
20            body = "{\"result\":\"success\",\"file\":\"/file/123.jpg\"}";
21        }else {
22            body = "{\"result\":\"faile\"}";
23        }
24        //打印结果
25        PrintWriter writer = response.getWriter();
26        writer.write(body);
27
28    }
29
30    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
31        this.doPost(request, response);
32    }
33}

我共使用了两种登录方式，分别是GET和POST这两种提交方式，

GET方式，一般登录不会用这种方式，GET是把数据放在地址上的，很容易就被其他人看到，我这只是测试而已

 1//使用GET方法
 2    public void get() {
 3        final String numberStr = number.getText().toString().trim();
 4        final String pwdStr = pwd.getText().toString().trim();
 5        //连接网络要使用线程
 6        new Thread(new Runnable() {
 7            @Override
 8            public void run() {
 9                //GET方法要在路径上写好数据
10                Request request = new Request.Builder()
11                        .url(url + "/MyServlet?number=" + numberStr + "&pwd=" + pwdStr)
12                        .build();
13                OkHttpClient client = new OkHttpClient.Builder()
14                        //连接超时
15                        .connectTimeout(10, TimeUnit.SECONDS)
16                        //写入超时
17                        .writeTimeout(20, TimeUnit.SECONDS)
18                        //读取超时
19                        .readTimeout(20, TimeUnit.SECONDS)
20                        .build();
21                Log.d("MainActivity","GET路径为："+request.url());
22                Call call = client.newCall(request);
23                call.enqueue(new Callback() {
24                    @Override
25                    public void onFailure(Call call, IOException e) {
26                        //失败做的一些处理
27                    }
28
29                    @Override
30                    public void onResponse(Call call, Response response) throws IOException {
31                        //获得返回数据
32                        String res = response.body().string();
33                        Log.d("MainActivity", "GET方法返回的数据为：" + res);
34                        parseJSON(res);
35                    }
36                });
37            }
38        }).start();
39    }

POST方式，这种方式用处比较多，比如上传文件也是用到POST方式

 1//使用POST方法
 2    public void post() {
 3        final String numberStr = number.getText().toString().trim();
 4        final String pwdStr = pwd.getText().toString().trim();
 5        //连接网络要使用线程
 6        new Thread(new Runnable() {

因为返回的数据是json格式，创建一个用于解析json的方法

 1    //解析JSON数据
 2    public void parseJSON(String jsonStr) {
 3        try {
 4            JSONObject jsonObject = new JSONObject(jsonStr);
 5            String resulet = jsonObject.getString("result");
 6            //判断是否成功
 7            if (resulet.equals("success")) {
 8                String fileUrl = jsonObject.getString("file");
 9                final String imageUrl = url + fileUrl;
10
11                //操作UI不能在子线程上操作，要在UI线程上操作
12                runOnUiThread(new Runnable() {
13                    @Override
14                    public void run() {
15                        //登录成功，跳转到另一个Activity
16                        Intent intent = new Intent(MainActivity.this, Main2Activity.class);
17                        intent.putExtra("imageUrl", imageUrl);
18                        startActivity(intent);
19
20                        Toast.makeText(getApplicationContext(), "登录成功", Toast.LENGTH_SHORT).show();
21                    }
22                });
23            } else {
24                //登录失败
25                runOnUiThread(new Runnable() {
26                    @Override
27                    public void run() {
28                        Toast.makeText(getApplicationContext(), "账号或密码错误", Toast.LENGTH_SHORT).show();
29                    }
30                });
31            }
32        } catch (JSONException e) {
33            e.printStackTrace();
34        }
35    }

显示网络图片的操作

 1private void showImage(final String ImageUrl) {
 2        //连接网络要使用线程
 3        new Thread(new Runnable() {
 4            @Override
 5            public void run() {
 6                try {
 7                    OkHttpClient client = new OkHttpClient.Builder()
 8                            //使用缓存
 9                            .cache(new Cache(getExternalCacheDir().getAbsoluteFile(), 1024 * 1024 * 2))
10                            .build();
11                    Request request = new Request.Builder()
12                            .url(ImageUrl)
13                            .build();
14                    Log.d("Main2Activity","图片路径为："+request.url());
15                    Response response = client.newCall(request).execute();
16                    //获取输入流
17                    InputStream inputStream = response.body().byteStream();
18                    Bitmap bitmap = BitmapFactory.decodeStream(inputStream);
19                    //使用Hander线程修改UI
20                    Message msg = new Message();
21                    msg.what = 1;
22                    msg.obj = bitmap;
23                    handler.sendMessage(msg);
24
25                    inputStream.close();
26                } catch (IOException e) {
27                    e.printStackTrace();
28                }
29            }
30        }).start();
31    }

还有一个保存网络图片到本地的SD卡上

 1public void saveImage(final String imageUrl) {
 2        //连接网络要使用线程
 3        new Thread(new Runnable() {
 4            @Override
 5            public void run() {
 6                try {
 7                    OkHttpClient client = new OkHttpClient.Builder()
 8                            //使用缓存
 9                            .cache(new Cache(getExternalCacheDir().getAbsoluteFile(), 1024 * 1024 * 2))
10                            .build();
11                    Request request = new Request.Builder()
12                            .url(imageUrl)
13                            .build();
14                    Log.d("Main2Activity","图片路径为："+request.url());
15                    Response response = client.newCall(request).execute();
16                    //获取字节数据
17                    byte[] bytes = response.body().bytes();
18                    //获取文件名
19                    String[] urls = url.split("/");
20                    String filename = urls[urls.length - 1];
21                    //保存文件
22                    File file = new File(Environment.getExternalStorageDirectory(), filename);
23                    FileOutputStream fos = new FileOutputStream(file);
24                    fos.write(bytes);
25                    fos.flush();
26                    fos.close();
27                    //弹出Toast
28                    Message msg = new Message();
29                    msg.what = 2;
30                    msg.obj = "保存成功";
31                    handler.sendMessage(msg);
32                } catch (IOException e) {
33                    e.printStackTrace();
34                    Log.e("Main2Activity", "图片保存失败");
35                    Message msg = new Message();
36                    msg.what = 2;
37                    msg.obj = "图片保存失败";
38                    handler.sendMessage(msg);
39                }
40            }
41        }).start();
42    }

在上面两种方法中都可以看到我使用了Handle更新UI，下面就是Handle的实现

 1 private Handler handler = new Handler() {
 2        @Override
 3        public void handleMessage(Message msg) {
 4            switch (msg.what) {
 5                case 1:
 6                    //设置图片
 7                    imageView.setImageBitmap((Bitmap) msg.obj);
 8                    break;
 9                case 2:
10                    //一些提示
11                    Toast.makeText(getApplicationContext(), (String) msg.obj, Toast.LENGTH_SHORT).show();
12                    break;
13            }
14        }
15    };
