生成图像列表

所以我们要编写一个CreateDataList.py程序可以为我们生成这样的图像列表
在这个程序中,我们只要把一个大类的文件夹路径传进去就可以了,该程序会把里面的每个小类别都迭代,生成固定格式的列表.比如我们把蔬菜类别的根目录传进去../images/vegetables

 1# coding=utf-8
 2import os
 3import json
 4
 5class CreateDataList:
 6    def __init__(self):
 7        pass
 8
 9    def createDataList(self, data_root_path):
10        # # 把生产的数据列表都放在自己的总类别文件夹中
11        data_list_path = ''
12        # 所有类别的信息
13        class_detail = []
14        # 获取所有类别
15        class_dirs = os.listdir(data_root_path)
16        # 类别标签
17        class_label = 0
18        # 获取总类别的名称
19        father_paths = data_root_path.split('/')
20        while True:
21            if father_paths[father_paths.__len__() - 1] == '':
22                del father_paths[father_paths.__len__() - 1]
23            else:
24                break
25        father_path = father_paths[father_paths.__len__() - 1]
26
27        all_class_images = 0
28        # 读取每个类别
29        for class_dir in class_dirs:
30            # 每个类别的信息
31            class_detail_list = {}
32            test_sum = 0
33            trainer_sum = 0
34            # 把生产的数据列表都放在自己的总类别文件夹中
35            data_list_path = "../data/%s/" % father_path
36            # 统计每个类别有多少张图片
37            class_sum = 0
38            # 获取类别路径
39            path = data_root_path + "/" + class_dir
40            # 获取所有图片
41            img_paths = os.listdir(path)
42            for img_path in img_paths:
43                # 每张图片的路径
44                name_path = path + '/' + img_path
45                # 如果不存在这个文件夹,就创建
46                isexist = os.path.exists(data_list_path)
47                if not isexist:
48                    os.makedirs(data_list_path)
49                # 每10张图片取一个做测试数据
50                if class_sum % 10 == 0:
51                    test_sum += 1
52                    with open(data_list_path + "test.list", 'a') as f:
53                        f.write(name_path + "\t%d" % class_label + "\n")
54                else:
55                    trainer_sum += 1
56                    with open(data_list_path + "trainer.list", 'a') as f:
57                        f.write(name_path + "\t%d" % class_label + "\n")
58                class_sum += 1
59                all_class_images += 1
60            class_label += 1
61            # 说明的json文件的class_detail数据
62            class_detail_list['class_name'] = class_dir
63            class_detail_list['class_label'] = class_label
64            class_detail_list['class_test_images'] = test_sum
65            class_detail_list['class_trainer_images'] = trainer_sum
66            class_detail.append(class_detail_list)
67        # 获取类别数量
68        all_class_sum = class_dirs.__len__()
69        # 说明的json文件信息
70        readjson = {}
71        readjson['all_class_name'] = father_path
72        readjson['all_class_sum'] = all_class_sum
73        readjson['all_class_images'] = all_class_images
74        readjson['class_detail'] = class_detail
75        jsons = json.dumps(readjson, sort_keys=True, indent=4, separators=(',', ': '))
76        with open(data_list_path + "readme.json",'w') as f:
77            f.write(jsons)
78
79
80if __name__ == '__main__':
81    createDataList = CreateDataList()
82    createDataList.createDataList('../images/vegetables')

读取数据

通过MyReader.py这个程序可以将上一部分的图像列表读取,生成训练和测试使用的reader,在生成reader前,要传入一个图像的大小,PaddlePaddle会帮我们按照这个大小随机裁剪一个方形的图像,这是种随机裁剪也是数据增强的一种方式.

 1from multiprocessing import cpu_count
 2import paddle.v2 as paddle
 3
 4class MyReader:
 5    def __init__(self,imageSize):
 6        self.imageSize = imageSize
 7
 8    def train_mapper(self,sample):
 9        '''
10        map image path to type needed by model input layer for the training set
11        '''
12        img, label = sample
13        img = paddle.image.load_image(img)
14        img = paddle.image.simple_transform(img, 70, self.imageSize, True)
15        return img.flatten().astype('float32'), label
16
17    def test_mapper(self,sample):
18        '''
19        map image path to type needed by model input layer for the test set
20        '''
21        img, label = sample
22        img = paddle.image.load_image(img)
23        img = paddle.image.simple_transform(img, 70, self.imageSize, False)
24        return img.flatten().astype('float32'), label
25
26    def train_reader(self,train_list, buffered_size=1024):
27        def reader():
28            with open(train_list, 'r') as f:
29                lines = [line.strip() for line in f]
30                for line in lines:
31                    img_path, lab = line.strip().split('\t')
32                    yield img_path, int(lab)
33
34        return paddle.reader.xmap_readers(self.train_mapper, reader,
35                                          cpu_count(), buffered_size)
36
37    def test_reader(self,test_list, buffered_size=1024):
38        def reader():
39            with open(test_list, 'r') as f:
40                lines = [line.strip() for line in f]
41                for line in lines:
42                    img_path, lab = line.strip().split('\t')
43                    yield img_path, int(lab)
44
45        return paddle.reader.xmap_readers(self.test_mapper, reader,
46                                          cpu_count(), buffered_size)

定义神经网络

编写一个vgg.py来定义VGG神经网络，这里使用的是VGG神经网络,跟上一篇文章用到的VGG又有一点不同,这里可以看到conv_with_batchnorm=False，我是把BN关闭了，这是因为启用BN层的同时，也会使用Dropout层，因为数据集比较小，再使用Dropout就更小了，导致模型无法收敛。如果读者一定要启动BN层的话，可以单独关闭Dropout，把drop_rate全部设置为0。如果数据集大的话，就可以不用这样处理。

 1# coding:utf-8
 2import paddle.v2 as paddle
 3
 4def vgg_bn_drop(datadim, type_size):
 5    # 获取输入数据模式
 6    image = paddle.layer.data(name="image",
 7                              type=paddle.data_type.dense_vector(datadim))
 8
 9    def conv_block(ipt, num_filter, groups, dropouts, num_channels=None):
10        return paddle.networks.img_conv_group(
11            input=ipt,
12            num_channels=num_channels,
13            pool_size=2,
14            pool_stride=2,
15            conv_num_filter=[num_filter] * groups,
16            conv_filter_size=3,
17            conv_act=paddle.activation.Relu(),
18            conv_with_batchnorm=False,
19            conv_batchnorm_drop_rate=dropouts,
20            pool_type=paddle.pooling.Max())
21
22    conv1 = conv_block(image, 64, 2, [0.3, 0], 3)
23    conv2 = conv_block(conv1, 128, 2, [0.4, 0])
24    conv3 = conv_block(conv2, 256, 3, [0.4, 0.4, 0])
25    conv4 = conv_block(conv3, 512, 3, [0.4, 0.4, 0])
26    conv5 = conv_block(conv4, 512, 3, [0.4, 0.4, 0])
27
28    drop = paddle.layer.dropout(input=conv5, dropout_rate=0.5)
29    fc1 = paddle.layer.fc(input=drop, size=512, act=paddle.activation.Linear())
30    bn = paddle.layer.batch_norm(input=fc1,
31                                 act=paddle.activation.Relu(),
32                                 layer_attr=paddle.attr.Extra(drop_rate=0.5))
33    fc2 = paddle.layer.fc(input=bn, size=512, act=paddle.activation.Linear())
34    # 通过Softmax获得分类器
35    out = paddle.layer.fc(input=fc2,
36                          size=type_size,
37                          act=paddle.activation.Softmax())
38    return out

获取参数

该函数可以通过输入是否是参数文件路径,或者是损失函数,如果是参数文件路径,就使用之前训练好的参数生产参数.如果不传入参数文件路径,那就使用传入的损失函数生成参数

 1def get_parameters(self, parameters_path=None, cost=None):
 2    if not parameters_path:
 3        # 使用cost创建parameters
 4        if not cost:
 5            raise NameError('请输入cost参数')
 6        else:
 7            # 根据损失函数创建参数
 8            parameters = paddle.parameters.create(cost)
 9            print "cost"
10            return parameters
11    else:
12        # 使用之前训练好的参数
13        try:
14            # 使用训练好的参数

16                parameters = paddle.parameters.Parameters.from_tar(f)
17            print "使用parameters"
18            return parameters
19        except Exception as e:
20            raise NameError("你的参数文件错误,具体问题是:%s" % e)

创建训练器

创建训练器要3个参数,分别是损失函数,参数,优化方法.通过图像的标签信息和分类器生成损失函数.参数可以选择是使用之前训练好的参数,然后在此基础上再进行训练,又或者是使用损失函数生成初始化参数.然后再生成优化方法.就可以创建一个训练器了.

 1# datadim 数据大小
 2def get_trainer(self, datadim, type_size, parameters_path):
 3    # 获得图片对于的信息标签
 4    label = paddle.layer.data(name="label",
 5                              type=paddle.data_type.integer_value(type_size))
 6
 7    # 获取全连接层,也就是分类器
 8    out = vgg_bn_drop(datadim=datadim, type_size=type_size)
 9
10    # 获得损失函数
11    cost = paddle.layer.classification_cost(input=out, label=label)
12
13    # 获得参数
14    if not parameters_path:
15        parameters = self.get_parameters(cost=cost)
16    else:
17        parameters = self.get_parameters(parameters_path=parameters_path)
18
19    '''
20    定义优化方法
21    learning_rate 迭代的速度
22    momentum 跟前面动量优化的比例
23    regularzation 正则化,防止过拟合
24    '''
25    optimizer = paddle.optimizer.Momentum(
26        momentum=0.9,
27        regularization=paddle.optimizer.L2Regularization(rate=0.0005 * 128),
28        learning_rate=0.001 / 128,
29        learning_rate_decay_a=0.1,
30        learning_rate_decay_b=128000 * 35,
31        learning_rate_schedule="discexp", )
32
33    '''
34    创建训练器
35    cost 分类器
36    parameters 训练参数,可以通过创建,也可以使用之前训练好的参数
37    update_equation 优化方法
38    '''
39    trainer = paddle.trainer.SGD(cost=cost,
40                                 parameters=parameters,
41                                 update_equation=optimizer)
42    return trainer

开始训练

要启动训练要4个参数,分别是训练数据,训练的轮数,训练过程中的事件处理,输入数据和标签的对应关系.
训练数据:这次的训练数据是我们自定义的数据集.
训练轮数:表示我们要训练多少轮,次数越多准确率越高,最终会稳定在一个固定的准确率上.不得不说的是这个会比MNIST数据集的速度慢很多
事件处理:训练过程中的一些事件处理,比如会在每个batch打印一次日志,在每个pass之后保存一下参数和测试一下测试数据集的预测准确率.
输入数据和标签的对应关系:说明输入数据是第0维度,标签是第1维度

 1# ***********************开始训练***************************************
 2def start_trainer(self, trainer, num_passes, save_parameters_name, trainer_reader, test_reader):
 3    # 获得数据
 4    reader = paddle.batch(reader=paddle.reader.shuffle(reader=trainer_reader,
 5                                                       buf_size=50000),
 6                          batch_size=128)
 7    # 保证保存模型的目录是存在的
 8    father_path = save_parameters_name[:save_parameters_name.rfind("/")]
 9    if not os.path.exists(father_path):
10        os.makedirs(father_path)
11
12    # 指定每条数据和padd.layer.data的对应关系
13    feeding = {"image": 0, "label": 1}
14
15    # 定义训练事件
16    def event_handler(event):
17        if isinstance(event, paddle.event.EndIteration):
18            if event.batch_id % 100 == 0:
19                print "\nPass %d, Batch %d, Cost %f, Error %s" % (
20                    event.pass_id, event.batch_id, event.cost, event.metrics['classification_error_evaluator'])
21            else:
22                sys.stdout.write('.')
23                sys.stdout.flush()
24
25        # 每一轮训练完成之后
26        if isinstance(event, paddle.event.EndPass):
27            # 保存训练好的参数
28            with open(save_parameters_name, 'w') as f:
29                trainer.save_parameter_to_tar(f)
30
31            # 测试准确率
32            result = trainer.test(reader=paddle.batch(reader=test_reader,
33                                                      batch_size=128),
34                                  feeding=feeding)
35            print "\nTest with Pass %d, Classification_Error %s" % (
36            event.pass_id, result.metrics['classification_error_evaluator'])
37
38    '''
39    开始训练
40    reader 训练数据
41    num_passes 训练的轮数
42    event_handler 训练的事件,比如在训练的时候要做一些什么事情
43    feeding 说明每条数据和padd.layer.data的对应关系
44    '''
45    trainer.train(reader=reader,
46                  num_passes=num_passes,
47                  event_handler=event_handler,
48                  feeding=feeding)

定义预测函数，该函数需要输入3个参数,
第一个是需要预测的图像,图像传入之后,会经过load_image函数处理,大小会变成32*32大小,训练是输入数据的大小一样.
第二个就是训练好的参数
第三个是通过神经模型生成的分类器

 1def to_prediction(image_paths, parameters, out, imageSize):
 2
 3    # 获得要预测的图片
 4    test_data = []
 5    for image_path in image_paths:
 6        test_data.append((paddle.image.load_and_transform(image_path, 70, imageSize, False)
 7                          .flatten().astype('float32'),))
 8
 9    # 获得预测结果
10    probs = paddle.infer(output_layer=out,
11                         parameters=parameters,
12                         input=test_data)
13    # 处理预测结果
14    lab = np.argsort(-probs)
15    # 返回概率最大的值和其对应的概率值
16    all_result = []
17    for i in range(0, lab.__len__()):
18        all_result.append([lab[i][0], probs[i][(lab[i][0])]])
19    return all_result

  9class PaddleUtil:
 10    # ***********************初始化操作***************************************
 11    def __init__(self):
 12        # 初始化paddpaddle,只是用CPU,把GPU关闭
 13        paddle.init(use_gpu=False, trainer_count=2)
 14
 15    # **********************获取参数***************************************
 16    def get_parameters(self, parameters_path=None, cost=None):
 17        if not parameters_path:
 18            # 使用cost创建parameters
 19            if not cost:
 20                raise NameError('请输入cost参数')
 21            else:
 22                # 根据损失函数创建参数
 23                parameters = paddle.parameters.create(cost)
 24                print "cost"
 25                return parameters
 26        else:
 27            # 使用之前训练好的参数
 28            try:
 29                # 使用训练好的参数
 30                with open(parameters_path, 'r') as f:
 31                    parameters = paddle.parameters.Parameters.from_tar(f)
 32                print "使用parameters"
 33                return parameters
 34            except Exception as e:
 35                raise NameError("你的参数文件错误,具体问题是:%s" % e)
 36
 37    # ***********************获取训练器***************************************
 38    # datadim 数据大小
 39    def get_trainer(self, datadim, type_size, parameters_path):
 40        # 获得图片对于的信息标签
 41        label = paddle.layer.data(name="label",
 42                                  type=paddle.data_type.integer_value(type_size))
 43
 44        # 获取全连接层,也就是分类器
 45        out = vgg_bn_drop(datadim=datadim, type_size=type_size)
 46
 47        # 获得损失函数
 48        cost = paddle.layer.classification_cost(input=out, label=label)
 49
 50        # 获得参数
 51        if not parameters_path:
 52            parameters = self.get_parameters(cost=cost)
 53        else:
 54            parameters = self.get_parameters(parameters_path=parameters_path)
 55
 56        '''
 57        定义优化方法
 58        learning_rate 迭代的速度
 59        momentum 跟前面动量优化的比例
 60        regularzation 正则化,防止过拟合
 61        '''
 62        optimizer = paddle.optimizer.Momentum(
 63            momentum=0.9,
 64            regularization=paddle.optimizer.L2Regularization(rate=0.0005 * 128),
 65            learning_rate=0.001 / 128,
 66            learning_rate_decay_a=0.1,
 67            learning_rate_decay_b=128000 * 35,
 68            learning_rate_schedule="discexp", )
 69
 70        '''
 71        创建训练器
 72        cost 分类器
 73        parameters 训练参数,可以通过创建,也可以使用之前训练好的参数
 74        update_equation 优化方法
 75        '''
 76        trainer = paddle.trainer.SGD(cost=cost,
 77                                     parameters=parameters,
 78                                     update_equation=optimizer)
 79        return trainer
 80
 81    # ***********************开始训练***************************************
 82    def start_trainer(self, trainer, num_passes, save_parameters_name, trainer_reader, test_reader):
 83        # 获得数据
 84        reader = paddle.batch(reader=paddle.reader.shuffle(reader=trainer_reader,
 85                                                           buf_size=50000),
 86                              batch_size=128)
 87        # 保证保存模型的目录是存在的
 88        father_path = save_parameters_name[:save_parameters_name.rfind("/")]
 89        if not os.path.exists(father_path):
 90            os.makedirs(father_path)
 91
 92        # 指定每条数据和padd.layer.data的对应关系
 93        feeding = {"image": 0, "label": 1}
 94
 95        # 定义训练事件
 96        def event_handler(event):
 97            if isinstance(event, paddle.event.EndIteration):
 98                if event.batch_id % 100 == 0:
 99                    print "\nPass %d, Batch %d, Cost %f, Error %s" % (
100                        event.pass_id, event.batch_id, event.cost, event.metrics['classification_error_evaluator'])
101                else:
102                    sys.stdout.write('.')
103                    sys.stdout.flush()
104
105            # 每一轮训练完成之后
106            if isinstance(event, paddle.event.EndPass):
107                # 保存训练好的参数
108                with open(save_parameters_name, 'w') as f:
109                    trainer.save_parameter_to_tar(f)
110
111                # 测试准确率
112                result = trainer.test(reader=paddle.batch(reader=test_reader,
113                                                          batch_size=128),
114                                      feeding=feeding)
115                print "\nTest with Pass %d, Classification_Error %s" % (
116                event.pass_id, result.metrics['classification_error_evaluator'])
117
118        '''
119        开始训练
120        reader 训练数据
121        num_passes 训练的轮数
122        event_handler 训练的事件,比如在训练的时候要做一些什么事情
123        feeding 说明每条数据和padd.layer.data的对应关系
124        '''
125        trainer.train(reader=reader,
126                      num_passes=num_passes,
127                      event_handler=event_handler,
128                      feeding=feeding)
129

139    parameters_path = "../model/model.tar"

 2import numpy as np
 3import paddle.v2 as paddle
 4
 5from vgg import vgg_bn_drop

 7
 8# **********************获取参数***************************************
 9def get_parameters(parameters_path):
10    with open(parameters_path, 'r') as f:
11        parameters = paddle.parameters.Parameters.from_tar(f)
12    return parameters

14
15# ***********************使用训练好的参数进行预测***************************************
16def to_prediction(image_paths, parameters, out, imageSize):
17    # 获得要预测的图片
18    test_data = []
19    for image_path in image_paths:
20        test_data.append((paddle.image.load_and_transform(image_path, 70, imageSize, False)
21                          .flatten().astype('float32'),))
22
23    # 获得预测结果
24    probs = paddle.infer(output_layer=out,
25                         parameters=parameters,
26                         input=test_data)
27    # 处理预测结果
28    lab = np.argsort(-probs)
29    # 返回概率最大的值和其对应的概率值
30    all_result = []
31    for i in range(0, lab.__len__()):
32        all_result.append([lab[i][0], probs[i][(lab[i][0])]])
33    return all_result

DownloadImages.py,下载图片的代码：
这个程序可以从百度图片中下载图片,可以多个类别一起下载,还可以指定下载数量

 1# -*- coding:utf-8 -*-
 2import re
 3import uuid
 4import requests
 5import os
 6
 7
 8class DownloadImages:
 9    def __init__(self,download_max,key_word):
10        self.download_sum = 0
11        self.download_max = download_max
12        self.key_word = key_word
13        self.save_path = '../images/download/' + key_word
14
15    def start_download(self):
16        self.download_sum = 0
17        gsm = 80
18        str_gsm = str(gsm)
19        pn = 0
20        if not os.path.exists(self.save_path):
21            os.makedirs(self.save_path)
22        while self.download_sum < self.download_max:
23            str_pn = str(self.download_sum)
24            url = 'http://image.baidu.com/search/flip?tn=baiduimage&ie=utf-8&' \
25                  'word=' + self.key_word + '&pn=' + str_pn + '&gsm=' + str_gsm + '&ct=&ic=0&lm=-1&width=0&height=0'
26            print url
27            result = requests.get(url)
28            self.downloadImages(result.text)
29        print '下载完成'
30
31    def downloadImages(self,html):
32        img_urls = re.findall('"objURL":"(.*?)",', html, re.S)
33        print '找到关键词:' + self.key_word + '的图片，现在开始下载图片...'
34        for img_url in img_urls:
35            print '正在下载第' + str(self.download_sum + 1) + '张图片，图片地址:' + str(img_url)
36            try:
37                pic = requests.get(img_url, timeout=50)
38                pic_name = self.save_path + '/' + str(uuid.uuid1()) + '.' + str(img_url).split('.')[-1]
39                with open(pic_name, 'wb') as f:
40                    f.write(pic.content)
41                self.download_sum += 1
42                if self.download_sum >= self.download_max:
43                    break
44            except  Exception, e:
45                print '【错误】当前图片无法下载，%s' % e
46                continue
